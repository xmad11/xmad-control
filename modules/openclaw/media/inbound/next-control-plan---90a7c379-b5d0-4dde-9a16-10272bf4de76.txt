"use client";

import { useState, useRef, useEffect } from "react";
import { Send, Bot, User } from "lucide-react";

interface Message {
  id: string;
  role: "user" | "nova";
  text: string;
  time: string;
}

export default function NovaPage() {
  const [messages, setMessages] = useState<Message[]>([
    {
      id: "0",
      role: "nova",
      text: "[👱Nova] Connected. What do you need?",
      time: new Date().toLocaleTimeString(),
    },
  ]);
  const [input, setInput] = useState("");
  const [sending, setSending] = useState(false);
  const bottomRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    bottomRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [messages]);

  async function send() {
    if (!input.trim() || sending) return;
    const text = input.trim();
    setInput("");
    setSending(true);

    const userMsg: Message = {
      id: Date.now().toString(),
      role: "user",
      text,
      time: new Date().toLocaleTimeString(),
    };
    setMessages((m) => [...m, userMsg]);

    try {
      const res = await fetch("/xmad-api/openclaw/message", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ message: text }),
      });
      const data = await res.json();
      const novaMsg: Message = {
        id: (Date.now() + 1).toString(),
        role: "nova",
        text: data.response || data.message || "No response",
        time: new Date().toLocaleTimeString(),
      };
      setMessages((m) => [...m, novaMsg]);
    } catch {
      setMessages((m) => [
        ...m,
        { id: (Date.now() + 1).toString(), role: "nova", text: "Error: could not reach Nova.", time: new Date().toLocaleTimeString() },
      ]);
    } finally {
      setSending(false);
    }
  }

  return (
    <div style={{ display: "flex", flexDirection: "column", height: "100vh", padding: 20 }}>
      <h2 style={{ fontSize: 16, fontWeight: 700, marginBottom: 16, color: "var(--text)" }}>Nova Chat</h2>

      {/* Messages */}
      <div
        style={{
          flex: 1,
          overflowY: "auto",
          background: "var(--surface)",
          border: "1px solid var(--border)",
          borderRadius: 10,
          padding: 16,
          display: "flex",
          flexDirection: "column",
          gap: 12,
          marginBottom: 12,
        }}
      >
        {messages.map((m) => (
          <div
            key={m.id}
            style={{
              display: "flex",
              gap: 10,
              flexDirection: m.role === "user" ? "row-reverse" : "row",
            }}
          >
            <div
              style={{
                width: 28,
                height: 28,
                borderRadius: "50%",
                background: m.role === "user" ? "var(--accent)" : "var(--accent2)",
                display: "flex",
                alignItems: "center",
                justifyContent: "center",
                flexShrink: 0,
              }}
            >
              {m.role === "user" ? <User size={14} /> : <Bot size={14} />}
            </div>
            <div
              style={{
                background: "var(--surface2)",
                border: "1px solid var(--border)",
                borderRadius: 8,
                padding: "8px 12px",
                maxWidth: "70%",
                fontSize: 13,
                lineHeight: 1.5,
                whiteSpace: "pre-wrap",
              }}
            >
              {m.text}
              <div style={{ color: "var(--text2)", fontSize: 10, marginTop: 4 }}>{m.time}</div>
            </div>
          </div>
        ))}
        {sending && (
          <div style={{ color: "var(--text2)", fontSize: 12, paddingLeft: 38 }}>Nova is thinking...</div>
        )}
        <div ref={bottomRef} />
      </div>

      {/* Input */}
      <div style={{ display: "flex", gap: 10 }}>
        <input
          value={input}
          onChange={(e) => setInput(e.target.value)}

          placeholder="Message Nova..."
          style={{
            flex: 1,
            background: "var(--surface)",
            border: "1px solid var(--border)",
            borderRadius: 8,
            padding: "10px 14px",
            color: "var(--text)",
            fontSize: 13,
            outline: "none",
          }}
        />

          onClick={send}
          disabled={sending}
          style={{
            background: "var(--accent)",
            border: "none",
            borderRadius: 8,
            padding: "10px 16px",
            color: "#fff",
            cursor: "pointer",
            display: "flex",
            alignItems: "center",
            gap: 6,
          }}
        >
          <Send size={14} />
        </button>
      </div>
    </div>
  );
}

"use client";

import { useState, useEffect } from "react";
import { Save, RotateCcw, FileText } from "lucide-react";

const BRAIN_FILES = [
  { label: "SOUL.md",     path: "/Users/ahmadabdullah/.openclaw/workspace/SOUL.md"     },
  { label: "MEMORY.md",   path: "/Users/ahmadabdullah/.openclaw/workspace/MEMORY.md"   },
  { label: "IDENTITY.md", path: "/Users/ahmadabdullah/.openclaw/workspace/IDENTITY.md" },
  { label: "Nova Identity", path: "/Users/ahmadabdullah/.openclaw/memory/nova-identity.md" },
];

export default function MemoryPage() {
  const [selected, setSelected] = useState(BRAIN_FILES[0]);
  const [content, setContent] = useState("");
  const [original, setOriginal] = useState("");
  const [saving, setSaving] = useState(false);
  const [msg, setMsg] = useState("");

  async function load(file: typeof BRAIN_FILES[0]) {
    setSelected(file);
    setContent("");
    setMsg("");
    try {
      const res = await fetch(`/xmad-api/file/read?path=${encodeURIComponent(file.path)}`);
      const data = await res.json();
      setContent(data.content ?? "");
      setOriginal(data.content ?? "");
    } catch {
      setContent("Error loading file.");
    }
  }

  async function save() {
    setSaving(true);
    try {
      const res = await fetch("/xmad-api/file/write", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ path: selected.path, content }),
      });
      const data = await res.json();
      if (data.ok) {
        setOriginal(content);
        setMsg("Saved ✓");
        setTimeout(() => setMsg(""), 3000);
      } else {
        setMsg("Save failed");
      }
    } catch {
      setMsg("Error saving");
    } finally {
      setSaving(false);
    }
  }

  useEffect(() => { load(BRAIN_FILES[0]); }, []);

  const dirty = content !== original;

  return (
    <div style={{ padding: 20, height: "100vh", display: "flex", flexDirection: "column" }}>
      <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center", marginBottom: 16 }}>
        <h2 style={{ fontSize: 16, fontWeight: 700 }}>Brain / Memory Editor</h2>
        <div style={{ display: "flex", gap: 8, alignItems: "center" }}>

        method: "POST",
        headers: { "Content-Type": "application/json" },

// lib/security.ts
// Path traversal prevention and input sanitization

import path from "node:path";
import { ALLOWED_MEMORY_PATHS, type AllowedMemoryPath } from "@/types/xmad";

/**
 * Validates that a requested file path is in the explicit allowlist.
 * No dynamic path resolution — only exact matches are permitted.
 * This prevents all path traversal attacks including:
 *   ../../../etc/passwd
 *   %2e%2e%2f (URL encoded)
 *   null byte injection
 */
export function validateMemoryPath(
  requested: string
): AllowedMemoryPath | null {
  // Normalize before comparing
  const normalized = path.normalize(requested);

  // Exact match only — no startsWith, no includes
  const match = ALLOWED_MEMORY_PATHS.find((p) => p === normalized);
  return match ?? null;
}

/**
 * Sanitize a string for use in shell commands.
 * Only used for display, never passed to exec.
 */
export function sanitizeForDisplay(input: string): string {
  return input
    .replace(/[<>'"&]/g, "")
    .slice(0, 2000);
}

/**
 * Check if a request is coming from a Tailscale IP range.
 * Tailscale uses 100.64.0.0/10 CGNAT range.
 */
export function isTailscaleIP(ip: string): boolean {
  if (ip === "127.0.0.1" || ip === "::1") return true; // localhost always allowed
  const parts = ip.split(".").map(Number);
  if (parts.length !== 4) return false;
  // 100.64.0.0 – 100.127.255.255
  return parts[0] === 100 && parts[1] >= 64 && parts[1] <= 127;
}

/**
 * Validate job task text — prevent injection into shell commands.
 * Tasks are passed to the Guardian orchestrator, never eval'd directly.
 */
export function validateTaskText(task: string): { ok: boolean; reason?: string } {
  if (!task || task.trim().length === 0) {
    return { ok: false, reason: "Task cannot be empty" };
  }
  if (task.length > 2000) {
    return { ok: false, reason: "Task too long (max 2000 chars)" };
  }
  // Block obvious injection attempts
  const forbidden = [";", "&&", "||", "`", "$(" , "$(", "<(", "|("];
  for (const f of forbidden) {
    if (task.includes(f)) {
      return { ok: false, reason: `Invalid character sequence: ${f}` };
    }
  }
  return { ok: true };
}

// lib/secrets.ts
// SSOT secrets loader — Keychain → Infisical → environment fallback
// Keys NEVER appear in code, logs, or frontend responses

/**
 * Load a secret using layered priority:
 * 1. macOS Keychain (via environment variable injected at process start by LaunchAgent)
 * 2. Environment variable (set by secrets manager or LaunchAgent)
 * 3. Explicit fallback (must be provided by caller — never hardcoded here)
 *
 * WHY NOT exec security(1) here:
 * Calling child_process.exec in a server component would block the render
 * and create a timing attack surface. Instead, the LaunchAgent script
 * pre-loads Keychain values into the process environment before Node starts.
 *
 * See: ~/xmad/scripts/load-secrets.sh
 */
export function getSecret(name: string, fallback?: string): string {
  const value = process.env[name];
  if (value && value.length > 0) return value;
  if (fallback !== undefined) return fallback;
  // Do NOT throw in production — log and return empty
  if (process.env.NODE_ENV !== "production") {
    console.warn(`[secrets] Missing secret: ${name}`);
  }
  return "";
}

// API client for XMAD Gateway (port 9870) with fallback to legacy (8765)
// All functions run server-side only — never import this in client components

import type {
  SystemStats,
  ServiceStatus,
  TailscaleStatus,
  AutomationJob,
  Backup,
  MemoryFile,
  AllowedMemoryPath,
} from "@/types/xmad";
import { getGatewayUrl, getLegacyApiUrl, getOpenClawUrl } from "@/lib/secrets";
import { validateMemoryPath } from "@/lib/security";

const TIMEOUT_MS = 8000;

async function fetchWithTimeout(
  url: string,
  options: RequestInit = {},
  timeoutMs = TIMEOUT_MS
): Promise<Response> {
  const controller = new AbortController();
  const id = setTimeout(() => controller.abort(), timeoutMs);
  try {
    const res = await fetch(url, { ...options, signal: controller.signal });
    return res;
  } finally {
    clearTimeout(id);
  }
}

/**
 * Try gateway first, fall back to legacy API.
 * Returns null on network failure — callers decide how to handle.
 */
async function gatewayFetch(
  path: string,
  options: RequestInit = {}
): Promise<{ data: unknown; source: "gateway" | "legacy" } | null> {
  const gatewayUrl = getGatewayUrl();
  const legacyUrl = getLegacyApiUrl();

  // Try gateway (9870)
  try {
    const res = await fetchWithTimeout(`${gatewayUrl}${path}`, {
      ...options,
      headers: {
        "Content-Type": "application/json",
        ...(options.headers ?? {}),
      },
      cache: "no-store",
    });
    if (res.ok) {
      return { data: await res.json(), source: "gateway" };
    }
  } catch {
    // Gateway not ready yet — fall through to legacy
  }

  // Fall back to legacy (8765)
  // Map gateway paths to legacy equivalents
  const legacyPath = mapToLegacyPath(path);
  try {
    const res = await fetchWithTimeout(`${legacyUrl}${legacyPath}`, {
      ...options,
      headers: {
        "Content-Type": "application/json",
        ...(options.headers ?? {}),
      },
      cache: "no-store",
    });
    if (res.ok) {
      return { data: await res.json(), source: "legacy" };
    }
  } catch {
    return null;
  }

  return null;
}

function mapToLegacyPath(gatewayPath: string): string {
  const map: Record<string, string> = {
    "/api/system/stats":       "/api/tools/run",

/**
 * Read a memory/brain file.
 * Path is validated against the allowlist before being returned.
 */
export async function readMemoryFile(
  requestedPath: string
): Promise<MemoryFile | { error: string }> {
  const safePath = validateMemoryPath(requestedPath);
  if (!safePath) {
    return { error: "Path not in allowlist" };
  }

  const result = await gatewayFetch(
    `/api/file/read?path=${encodeURIComponent(safePath)}`
  );
  if (!result) return { error: "Gateway unavailable" };

  const d = result.data as Record<string, unknown>;
  if (d.error) return { error: d.error as string };

  return {
    path:     safePath,
    label:    safePath.split("/").pop() ?? safePath,
    content:  (d.content as string) ?? "",
    sizeByte: (d.size as number) ?? 0,
    modified: (d.modified as string) ?? new Date().toISOString(),
  };
}

/**
 * Write a memory/brain file.
 * Path validated against allowlist — no dynamic paths accepted.
 */
export async function writeMemoryFile(
  requestedPath: string,
  content: string
): Promise<{ ok: boolean; error?: string }> {
  const safePath = validateMemoryPath(requestedPath);
  if (!safePath) {
    return { ok: false, error: "Path not in allowlist" };
  }

  if (content.length > 200_000) {
    return { ok: false, error: "Content too large (200KB max)" };
  }

  const result = await gatewayFetch("/api/file/write", {
    method: "POST",
    body: JSON.stringify({ path: safePath, content }),
  });

  if (!result) return { ok: false, error: "Gateway unavailable" };
  const d = result.data as { ok?: boolean; error?: string };
  return { ok: d.ok ?? false, error: d.error };
}

/**
 * Send a message to Nova via OpenClaw API.
 * Used by the chat page.
 */
export async function sendNovaMessage(
  message: string
): Promise<{ response: string } | { error: string }> {
  const openclawUrl = getOpenClawUrl();
  try {
    const res = await fetchWithTimeout(
      `${openclawUrl}/api/message`,
      {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ message, channel: "xmad-web" }),
        cache: "no-store",
      },
      30_000 // Nova can take a while
    );
    if (!res.ok) return { error: `OpenClaw HTTP ${res.status}` };
    const data = (await res.json()) as { response?: string; message?: string };
    return { response: data.response ?? data.message ?? "No response" };
  } catch (err) {
    return { error: err instanceof Error ? err.message : "Network error" };
  }

// Server-Sent Events utilities — server-side only

import type { SSEEvent } from "@/types/xmad";

/**
 * Create a ReadableStream that emits SSE-formatted events.
 * Used by /api/xmad/events route.
 */
export function createSSEStream(
  handler: (emit: (event: SSEEvent) => void, close: () => void) => () => void
): ReadableStream {
  let cleanup: (() => void) | null = null;

  return new ReadableStream({
    start(controller) {
      function emit(event: SSEEvent) {
        try {
          const line = `data: ${JSON.stringify(event)}\n\n`;
          controller.enqueue(new TextEncoder().encode(line));
        } catch {
          // Controller may be closed
        }
      }

      function close() {
        try {
          controller.close();
        } catch {
          // Already closed
        }
      }

      cleanup = handler(emit, close);

      // Heartbeat to keep connection alive through proxies
      const heartbeat = setInterval(() => {
        emit({
          type: "heartbeat",
          payload: { time: new Date().toISOString() },
          ts: new Date().toISOString(),
        });
      }, 25_000);

      // Store cleanup
      const originalCleanup = cleanup;
      cleanup = () => {
        clearInterval(heartbeat);
        if (originalCleanup) originalCleanup();
      };
    },
    cancel() {
      if (cleanup) cleanup();
    },
  });
}

/**
 * Format an SSE event string manually (for simple cases).
 */
export function formatSSE(event: SSEEvent): string {
  return `data: ${JSON.stringify(event)}\n\n`;
}

/**
 * Parse incoming SSE data string on the client.
 * Returns null if the data is a heartbeat.
 */
export function parseSSEData(raw: string): SSEEvent | null {
  try {
    const parsed = JSON.parse(raw) as SSEEvent;
    if (parsed.type === "heartbeat") return null;
    return parsed;
  } catch {
    return null;
  }
}

// hooks/use-sse.ts
"use client";

import { useEffect, useRef, useState, useCallback } from "react";
import type { SSEEvent } from "@/types/xmad";
import { parseSSEData } from "@/lib/sse";

interface UseSSEOptions {
  url: string;
  onEvent?: (event: SSEEvent) => void;
  reconnectMs?: number;
  maxRetries?: number;
}

interface UseSSEState {
  connected: boolean;
  error: string | null;
  retryCount: number;
}

export function useSSE({
  url,
  onEvent,
  reconnectMs = 5000,
  maxRetries = 10,
}: UseSSEOptions): UseSSEState {
  const [state, setState] = useState<UseSSEState>({
    connected: false,
    error: null,
    retryCount: 0,
  });

  const esRef    = useRef<EventSource | null>(null);
  const retries  = useRef(0);
  const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null);
  const onEventRef = useRef(onEvent);
  onEventRef.current = onEvent;

  const connect = useCallback(() => {
    if (esRef.current) {
      esRef.current.close();
      esRef.current = null;
    }

    const es = new EventSource(url);
    esRef.current = es;

    es.onopen = () => {
      retries.current = 0;
      setState({ connected: true, error: null, retryCount: 0 });
    };

    es.onmessage = (ev) => {
      const parsed = parseSSEData(ev.data);
      if (parsed && onEventRef.current) {
        onEventRef.current(parsed);
      }
    };

    es.onerror = () => {
      es.close();
      esRef.current = null;
      setState((s) => ({ ...s, connected: false }));

      if (retries.current >= maxRetries) {
        setState((s) => ({
          ...s,
          error: `Lost connection after ${maxRetries} retries`,
        }));
        return;
      }

      retries.current++;
      setState((s) => ({ ...s, retryCount: retries.current }));

      timerRef.current = setTimeout(connect, reconnectMs);
    };
  }, [url, reconnectMs, maxRetries]);

  useEffect(() => {
    connect();
    return () => {
      if (timerRef.current) clearTimeout(timerRef.current);
      if (esRef.current) esRef.current.close();
    };
  }, [connect]);

  return state;
}

"use client";

import { useState, useEffect, useCallback } from "react";
import type { SystemStats, ServiceStatus } from "@/types/xmad";

interface UseSystemStatsResult {
  stats:    SystemStats | null;
  services: ServiceStatus[];
  loading:  boolean;
  refresh:  () => void;
}

export function useSystemStats(pollMs = 15_000): UseSystemStatsResult {
  const [stats,    setStats]    = useState<SystemStats | null>(null);
  const [services, setServices] = useState<ServiceStatus[]>([]);
  const [loading,  setLoading]  = useState(true);

  const fetch_ = useCallback(async () => {
    try {
      const [statsRes, svcRes] = await Promise.all([
        fetch("/api/xmad/system/stats", { cache: "no-store" }),
        fetch("/api/xmad/services",     { cache: "no-store" }),
      ]);

      if (statsRes.ok) {
        const d = await statsRes.json() as SystemStats;
        setStats(d);
      }
      if (svcRes.ok) {
        const d = await svcRes.json() as { services: ServiceStatus[] };
        setServices(d.services ?? []);
      }
    } catch {
      // Silent — components show stale data
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => {
    fetch_();
    const timer = setInterval(fetch_, pollMs);
    return () => clearInterval(timer);
  }, [fetch_, pollMs]);

  return { stats, services, loading, refresh: fetch_ };
}

// app/api/xmad/[...path]/route.ts
// Proxy: Next.js → XMAD Gateway (9870) with legacy fallback (8765)
// Runs server-side — secrets never reach the client

import { type NextRequest, NextResponse } from "next/server";
import { getGatewayUrl, getLegacyApiUrl } from "@/lib/secrets";

export const dynamic = "force-dynamic";

const TIMEOUT_MS = 10_000;

async function proxyRequest(
  req: NextRequest,
  targetBase: string,
  path: string
): Promise<Response | null> {
  const url = `${targetBase}/api/${path}`;
  const controller = new AbortController();
  const tid = setTimeout(() => controller.abort(), TIMEOUT_MS);

  try {
    const body = req.method !== "GET" && req.method !== "HEAD"
      ? await req.text()
      : undefined;

    const res = await fetch(url, {
      method: req.method,
      headers: {
        "Content-Type": "application/json",
        "X-Forwarded-For": req.headers.get("x-forwarded-for") ?? "",
      },
      body,
      signal: controller.signal,
    });

    return res;
  } catch {
    return null;
  } finally {
    clearTimeout(tid);
  }
}

export async function GET(
  req: NextRequest,
  { params }: { params: Promise<{ path: string[] }> }
) {
  const { path } = await params;
  const pathStr = path.join("/");

  // Try gateway first
  const gwRes = await proxyRequest(req, getGatewayUrl(), pathStr);
  if (gwRes?.ok) {
    const data = await gwRes.json();
    return NextResponse.json(data);
  }

  // Fall back to legacy
  const legacyRes = await proxyRequest(req, getLegacyApiUrl(), pathStr);
  if (legacyRes?.ok) {
    const data = await legacyRes.json();
    return NextResponse.json({ ...data, _source: "legacy" });
  }

  return NextResponse.json(
    { error: "Service unavailable", gateway: getGatewayUrl(), legacy: getLegacyApiUrl() },
    { status: 503 }
  );
}

export async function POST(
  req: NextRequest,
  { params }: { params: Promise<{ path: string[] }> }
) {
  return GET(req, { params });
}

export async function DELETE(
  req: NextRequest,
  { params }: { params: Promise<{ path: string[] }> }
) {
  return GET(req, { params });
}

// app/api/xmad/events/route.ts
// Server-Sent Events endpoint — pushes live system updates to the dashboard

import { type NextRequest, NextResponse } from "next/server";
import { createSSEStream } from "@/lib/sse";
import { getGatewayUrl, getLegacyApiUrl } from "@/lib/secrets";
import type { SSEEvent, SystemStats, ServiceStatus } from "@/types/xmad";

export const dynamic = "force-dynamic";

async function fetchStats(): Promise<SystemStats | null> {
  for (const base of [getGatewayUrl(), getLegacyApiUrl()]) {
    try {
      const res = await fetch(`${base}/api/system/stats`, {
        cache: "no-store",
        signal: AbortSignal.timeout(5000),
      });
      if (res.ok) return res.json() as Promise<SystemStats>;
    } catch { /* try next */ }
  }
  return null;
}

async function fetchServices(): Promise<ServiceStatus[]> {
  for (const base of [getGatewayUrl(), getLegacyApiUrl()]) {
    try {
      const res = await fetch(`${base}/api/services`, {
        cache: "no-store",
        signal: AbortSignal.timeout(5000),
      });
      if (res.ok) {
        const d = await res.json() as { services?: ServiceStatus[] };
        return d.services ?? [];
      }
    } catch { /* try next */ }
  }
  return [];
}

export async function GET(req: NextRequest) {
  const stream = createSSEStream((emit, close) => {
    let active = true;

    async function poll() {
      if (!active) return;

      const [stats, services] = await Promise.all([fetchStats(), fetchServices()]);

      if (stats) {
        emit({ type: "stats", payload: stats, ts: new Date().toISOString() } satisfies SSEEvent);
      }

      if (services.length > 0) {
        emit({ type: "service", payload: services, ts: new Date().toISOString() } satisfies SSEEvent);
      }

      // RAM alert
      if (stats && stats.memFreeMB < 300) {
        emit({
          type: "alert",
          payload: { level: "critical", message: `Free RAM critically low: ${stats.memFreeMB}MB` },
          ts: new Date().toISOString(),
        } satisfies SSEEvent);
      } else if (stats && stats.memFreeMB < 500) {
        emit({
          type: "alert",
          payload: { level: "warn", message: `Free RAM low: ${stats.memFreeMB}MB` },
          ts: new Date().toISOString(),
        } satisfies SSEEvent);
      }
    }

    // Initial push immediately
    poll();
    const timer = setInterval(poll, 12_000); // every 12s

    // Handle client disconnect
    req.signal.addEventListener("abort", () => {
      active = false;
      clearInterval(timer);
      close();
    });

    return () => {
      active = false;
      clearInterval(timer);
    };
  });

  return new NextResponse(stream, {
    headers: {
      "Content-Type":  "text/event-stream",
      "Cache-Control": "no-cache, no-transform",
      "Connection":    "keep-alive",
      "X-Accel-Buffering": "no", // Disable nginx buffering
    },
  });
}

// middleware.ts
// Blocks access from non-Tailscale, non-localhost IPs
// Tailscale CGNAT range: 100.64.0.0/10

import { type NextRequest, NextResponse } from "next/server";

function isTailscaleOrLocal(ip: string): boolean {
  if (!ip) return false;
  if (ip === "127.0.0.1" || ip === "::1" || ip === "localhost") return true;
  if (ip.startsWith("::ffff:127.")) return true; // IPv6-mapped IPv4 localhost

  const parts = ip.split(".").map(Number);
  if (parts.length !== 4 || parts.some(isNaN)) return false;

  // Tailscale range: 100.64.0.0 – 100.127.255.255
  return parts[0] === 100 && parts[1] >= 64 && parts[1] <= 127;
}

export function middleware(req: NextRequest) {
  // Skip in dev — only enforce in production
  if (process.env.NODE_ENV !== "production") {
    return NextResponse.next();
  }

  const forwarded = req.headers.get("x-forwarded-for");
  const realIP    = req.headers.get("x-real-ip");
  const ip        = (forwarded?.split(",")[0] ?? realIP ?? "").trim();

  if (!isTailscaleOrLocal(ip)) {
    return new NextResponse(
      JSON.stringify({
        error: "Access restricted to Tailscale network",
        yourIP: ip,
        hint:   "Connect via Tailscale VPN then retry",
      }),
      {
        status:  403,
        headers: { "Content-Type": "application/json" },
      }
    );
  }

  return NextResponse.next();
}

  // Apply to all routes except public assets
  matcher: ["/((?!_next/static|_next/image|favicon.ico).*)"],

FILE 32: hooks/use-alerts.ts — Alert deduplication

// hooks/use-alerts.ts
"use client";

import { useState, useCallback } from "react";

export interface Alert {
  id:      string;
  level:   "warn" | "critical";
  message: string;
  ts:      string;
}

export function useAlerts(maxAlerts = 5) {
  const [alerts, setAlerts] = useState<Alert[]>([]);

  const push = useCallback((message: string, level: "warn" | "critical" = "warn") => {
    setAlerts((prev) => {
      // Deduplicate — don't push the same message twice in 60 seconds
      const now = Date.now();
      const duplicate = prev.find(
        (a) => a.message === message && now - new Date(a.ts).getTime() < 60_000
      );
      if (duplicate) return prev;

      const newAlert: Alert = {
        id:      `${now}-${Math.random().toString(36).slice(2, 6)}`,
        level,
        message,
        ts:      new Date().toISOString(),
      };

      return [newAlert, ...prev].slice(0, maxAlerts);
    });
  }, [maxAlerts]);

  const dismiss = useCallback((id: string) => {
    setAlerts((prev) => prev.filter((a) => a.id !== id));
  }, []);

  const clear = useCallback(() => setAlerts([]), []);

  return { alerts, push, dismiss, clear };
}

type LogSource = "openclaw" | "watchdog" | "xmad" | "guardian";

const SOURCES: { id: LogSource; label: string }[] = [
  { id: "openclaw",  label: "Nova / OpenClaw" },
  { id: "watchdog",  label: "Watchdog"        },
  { id: "xmad",      label: "XMAD API"        },
  { id: "guardian",  label: "Guardian"         },
];

export default function TerminalPage() {
  const [source,   setSource]   = useState<LogSource>("openclaw");
  const [lines,    setLines]    = useState<string[]>([]);
  const [loading,  setLoading]  = useState(false);
  const [autoRefresh, setAutoRefresh] = useState(false);
  const bottomRef = useRef<HTMLDivElement>(null);

  const fetchLogs = useCallback(async () => {
    setLoading(true);
    try {
      const res = await fetch(`/api/xmad/logs/${source}?lines=200`);
      if (res.ok) {
        const d = await res.json() as { lines?: string[] };
        setLines(d.lines ?? []);
        setTimeout(() => bottomRef.current?.scrollIntoView({ behavior: "smooth" }), 50);
      }
    } catch { /* silent */ }
    setLoading(false);
  }, [source]);

  useEffect(() => {
    fetchLogs();
  }, [fetchLogs]);

  useEffect(() => {
    if (!autoRefresh) return;
    const t = setInterval(fetchLogs, 5000);
    return () => clearInterval(t);
  }, [autoRefresh, fetchLogs]);

  function download() {
    const blob = new Blob([lines.join("\n")], { type: "text/plain" });
    const url  = URL.createObjectURL(blob);
    const a    = document.createElement("a");
    a.href     = url;
    a.download = `${source}-${Date.now()}.log`;
    a.click();
    URL.revokeObjectURL(url);
  }

  return (
    <div style={{ display: "flex", flexDirection: "column", height: "100dvh", padding: 20, gap: 12 }}>
      <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center" }}>
        <h2 style={{ fontFamily: "Syne, sans-serif", fontSize: 16, fontWeight: 700, margin: 0, display: "flex", alignItems: "center", gap: 8 }}>
          <Terminal size={16} /> Logs
        </h2>
        <div style={{ display: "flex", gap: 8, alignItems: "center" }}>
          <Button
            size="sm"
            variant={autoRefresh ? "success" : "ghost"}
            onClick={() => setAutoRefresh((v) => !v)}
          >
            {autoRefresh ? "● Live" : "Live Off"}
          </Button>
          <Button size="sm" icon={<RefreshCw size={12} />} onClick={fetchLogs} loading={loading}>
            Refresh
          </Button>
          <Button size="sm" icon={<Download size={12} />} onClick={download} disabled={lines.length === 0}>
            Download
          </Button>
        </div>
      </div>

      {/* Source tabs */}
      <div style={{ display: "flex", gap: 6 }}>
        {SOURCES.map((s) => (
          <button
            key={s.id}
            onClick={() => setSource(s.id)}
            style={{
              padding: "6px 12px",
              borderRadius: 6,
              border: `1px solid ${source === s.id ? "rgba(108,99,255,0.3)" : "var(--border)"}`,
              background: source === s.id ? "rgba(108,99,255,0.12)" : "transparent",
              color: source === s.id ? "var(--accent)" : "var(--text2)",
              cursor: "pointer",
              fontSize: 11,
              fontFamily: "inherit",
              fontWeight: source === s.id ? 700 : 400,
              transition: "all 0.15s",
            }}
          >
            {s.label}
          </button>
        ))}
      </div>

      {/* Log output */}
      <GlassCard
        style={{
          flex: 1,
          padding: 16,
          overflow: "auto",
          fontFamily: "Space Mono, monospace",
          fontSize: 11,
          lineHeight: 1.7,
        }}
      >
        {lines.length === 0 && !loading && (
          <div style={{ color: "var(--text2)", textAlign: "center", paddingTop: 40 }}>
            No log lines found for {source}.
          </div>
        )}
        {lines.map((line, i) => {
          const isError = /error|fail|crash|kill|timeout/i.test(line);
          const isWarn  = /warn|retry|reconnect/i.test(line);
          const isOk    = /started|success|online|connected|ready/i.test(line);
          return (
            <div
              key={i}
              style={{
                color: isError ? "var(--danger)" : isWarn ? "var(--warn)" : isOk ? "var(--ok)" : "var(--text2)",
                padding: "1px 0",
                whiteSpace: "pre-wrap",
                wordBreak: "break-all",
              }}
            >
              {line}
            </div>
          );
        })}
        <div ref={bottomRef} />
      </GlassCard>
    </div>
  );
}

// app/tasks/page.tsx
// Local task tracker — stored in localStorage, no backend required
"use client";

import { useState, useEffect } from "react";
import { GlassCard } from "@/components/ui/glass-card";
import { Button } from "@/components/ui/button";
import { Plus, Check, Trash2, ListTodo } from "lucide-react";

interface Task {
  id:        string;
  text:      string;
  done:      boolean;
  created:   string;
  priority:  "low" | "normal" | "high";
}

export default function TasksPage() {
  const [tasks,    setTasks]    = useState<Task[]>([]);
  const [newTask,  setNewTask]  = useState("");
  const [priority, setPriority] = useState<Task["priority"]>("normal");
  const [filter,   setFilter]   = useState<"all" | "active" | "done">("all");

  useEffect(() => {
    try {
      const raw = localStorage.getItem("xmad-tasks-v2");
      if (raw) setTasks(JSON.parse(raw));
    } catch { /* ignore */ }
  }, []);

  function persist(updated: Task[]) {
    setTasks(updated);
    localStorage.setItem("xmad-tasks-v2", JSON.stringify(updated));
  }

  function add() {
    if (!newTask.trim()) return;
    const task: Task = {
      id:       Date.now().toString(),
      text:     newTask.trim(),
      done:     false,
      created:  new Date().toISOString(),
      priority,
    };
    persist([task, ...tasks]);
    setNewTask("");
  }

  function toggle(id: string) {
    persist(tasks.map((t) => t.id === id ? { ...t, done: !t.done } : t));
  }

  function remove(id: string) {
    persist(tasks.filter((t) => t.id !== id));
  }

  const filtered = tasks.filter((t) => {
    if (filter === "active") return !t.done;
    if (filter === "done")   return t.done;
    return true;
  });

  const PRIORITY_COLOR: Record<Task["priority"], string> = {
    low:    "var(--text2)",
    normal: "var(--accent)",
    high:   "var(--danger)",
  };

  return (
    <div style={{ padding: 20 }}>
      <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center", marginBottom: 20 }}>
        <h2 style={{ fontFamily: "Syne, sans-serif", fontSize: 16, fontWeight: 700, margin: 0, display: "flex", alignItems: "center", gap: 8 }}>
          <ListTodo size={16} /> Tasks
        </h2>
        <div style={{ color: "var(--text2)", fontSize: 11 }}>
          {tasks.filter((t) => !t.done).length} active · {tasks.filter((t) => t.done).length} done
        </div>
      </div>

      {/* Add task */}
      <GlassCard style={{ marginBottom: 16 }}>
        <div style={{ display: "flex", gap: 10, flexWrap: "wrap" }}>
          <input
            value={newTask}
            onChange={(e) => setNewTask(e.target.value)}
            onKeyDown={(e) => e.key === "Enter" && add()}
            placeholder="New task..."
            style={{
              flex: 1, minWidth: 200,
              background: "rgba(255,255,255,0.03)",
              border: "1px solid var(--border)",
              borderRadius: 8,
              padding: "10px 14px",
              color: "var(--text)",
              fontSize: 13,
              outline: "none",
              fontFamily: "inherit",
            }}
          />
          <select
            value={priority}
            onChange={(e) => setPriority(e.target.value as Task["priority"])}
            style={{
              background: "rgba(255,255,255,0.03)",
              border: "1px solid var(--border)",
              borderRadius: 8,
              padding: "10px 12px",
              color: PRIORITY_COLOR[priority],
              fontSize: 12,
              outline: "none",
              fontFamily: "inherit",
              cursor: "pointer",
            }}
          >
            <option value="low">Low</option>
            <option value="normal">Normal</option>
            <option value="high">High</option>
          </select>
          <Button variant="primary" icon={<Plus size={14} />} onClick={add}>Add</Button>
        </div>
      </GlassCard>

      {/* Filter tabs */}
      <div style={{ display: "flex", gap: 6, marginBottom: 12 }}>
        {(["all", "active", "done"] as const).map((f) => (
          <button
            key={f}
            onClick={() => setFilter(f)}
            style={{
              padding: "5px 12px",
              borderRadius: 6,
              border: `1px solid ${filter === f ? "rgba(108,99,255,0.3)" : "var(--border)"}`,
              background: filter === f ? "rgba(108,99,255,0.12)" : "transparent",
              color: filter === f ? "var(--accent)" : "var(--text2)",
              cursor: "pointer",
              fontSize: 11,
              fontFamily: "inherit",
              fontWeight: filter === f ? 700 : 400,
              textTransform: "capitalize",
            }}
          >
            {f}
          </button>
        ))}
      </div>

      {/* Tasks */}
      <div style={{ display: "flex", flexDirection: "column", gap: 6 }}>
        {filtered.length === 0 && (
          <GlassCard style={{ textAlign: "center", padding: 32, color: "var(--text2)" }}>
            No {filter === "all" ? "" : filter} tasks.
          </GlassCard>
        )}
        {filtered.map((task) => (
          <GlassCard key={task.id} style={{ padding: 12 }}>
            <div style={{ display: "flex", alignItems: "center", gap: 10 }}>
              <button
                onClick={() => toggle(task.id)}
                style={{
                  width: 20, height: 20,
                  borderRadius: 6,
                  border: `2px solid ${task.done ? "var(--ok)" : "var(--border)"}`,
                  background: task.done ? "var(--ok)" : "transparent",
                  cursor: "pointer",
                  display: "flex",
                  alignItems: "center",
                  justifyContent: "center",
                  flexShrink: 0,
                  transition: "all 0.15s",
                }}
              >
                {task.done && <Check size={11} color="#000" />}
              </button>
              <div style={{ flex: 1, minWidth: 0 }}>
                <span
                  style={{
                    fontSize: 13,
                    color: task.done ? "var(--text2)" : "var(--text)",
                    textDecoration: task.done ? "line-through" : "none",
                    opacity: task.done ? 0.5 : 1,
                  }}
                >
                  {task.text}
                </span>
              </div>
              <div
                style={{
                  width: 6, height: 6, borderRadius: "50%",
                  background: PRIORITY_COLOR[task.priority],
                  flexShrink: 0,
                }}
              />
              <button
                onClick={() => remove(task.id)}
                style={{
                  background: "none",
                  border: "none",
                  cursor: "pointer",
                  color: "var(--text2)",
                  opacity: 0.5,
                  padding: 4,
                  flexShrink: 0,
                }}
              >
                <Trash2 size={12} />
              </button>
            </div>
          </GlassCard>
        ))}
      </div>
    </div>
  );
}


Recommendations and Enhancements Summary
Beyond what the plan specified, here is what I added and why each one matters for your real-world setup:
SSE RAM alerting in real-time. The SSE route in FILE 15 checks free RAM on every poll and fires a critical or warn event directly to the dashboard. The dashboard FILE 20 shows these as dismissable banners. This would have caught last week’s crash — you’d have seen the warning before OpenClaw went into I/O hang.
Middleware IP guard. FILE 16 blocks any request not from a Tailscale IP or localhost in production mode. No configuration needed — it checks the 100.64.0.0/10 CGNAT range that Tailscale always uses. In dev mode it’s disabled so local development still works normally.
Secrets loaded before Node starts. FILE 29 is a shell script that pulls from macOS Keychain and injects into the process environment before bun start runs. The LaunchAgent in FILE 30 calls this script instead of calling bun directly. This means process.env.ZAI_API_KEY is always populated without any secret ever touching the filesystem or a .env file.
Alert deduplication hook. FILE 32 prevents the same RAM warning from appearing 20 times in 5 minutes. If the same message was shown within 60 seconds, it’s silently dropped. This keeps the UI clean during the kind of repeated-poll situation that happens when OpenClaw is struggling.
Read-only log terminal. FILE 33 is a log viewer, not a shell executor. It reads log files through the API and color-codes lines by severity. This is an important security boundary — the terminal page cannot execute commands, only read logs. Live refresh can be toggled on for watching Nova restart in real time.
Gateway fallback is automatic. Every API call in lib/xmad-api.ts tries port 9870 first and falls back to 8765 silently. The response includes a _source: "legacy" field when the fallback is used. This means you can build the new gateway at your own pace without breaking the dashboard.​​​​​​​​​​​​​​​​

1. API proxy at /api/xmad/[...path]

   → Only enforced when NODE_ENV=production

   → load-secrets.sh pulls from macOS Keychain at process start
   → Values land in process.env before Node initializes
   → lib/secrets.ts reads process.env — no file reading at runtime
   → If a key is missing, the function returns "" and logs a warning in dev only

6. Button component has no form element

   → This is required for the artifact rendering environment

Here’s a full breakdown of an optimal architecture to keep OpenClaw (and the full AI Ops dashboard) under ~400MB RAM, while keeping everything stable and responsive.

⸻

1. Core Principles
	1.	Process Isolation
	•	Run each heavy service in its own worker process or container.
	•	Example:

OpenClaw Core → Worker Process
Automation Queue → Worker Process
Memory Index → Worker Process


	•	Prevents a single service from consuming all RAM.

	2.	Lazy Loading
	•	Only load modules when needed.
	•	Example:
	•	AI models in AI Chat are loaded per request.
	•	Memory editor loads only the segment being edited.
	3.	Streaming Instead of Full Buffering
	•	Logs, events, VNC frames, and API responses should stream, not load all into memory.
	•	Use Node.js streams or SSE/EventSource for logs and events.
	4.	Memory-efficient Caching
	•	Use LRU caches for AI memory, vector embeddings, and system snapshots.
	•	Limit each cache:

maxEntries: 500
maxSizeMB: 50
eviction: oldest


	5.	Garbage Collection and Periodic Cleanup
	•	For long-running processes, schedule:
	•	Memory trimming (global.gc() with --expose-gc)
	•	Event queue pruning
	•	Snapshot cleanup

⸻

2. Recommended Module Architecture (with RAM constraints)

Module	Strategy to Save RAM	Notes
OpenClaw Core	Worker process, lazy model load	Keep in ~150MB
AI Chat	Load models on demand, unload after response	Use smaller models for live interactions
Memory Editor	Load partial memory chunks, streaming edits	Keep only active chunk in RAM
Automation Queue	Worker pool, limit concurrent jobs	2-3 workers max, queue in DB/files
Backup Manager	Stream backups to disk/cloud	Do not keep full snapshots in RAM
Logs/Event Streams	SSE / streaming, cap lines in memory	Keep last 200 lines, older in storage
VNC Viewer	Proxy only, no full desktop load	Stream compressed frames
Workflow Scheduler	Lightweight in-memory scheduler	Persist tasks to disk/db
Task Executors	Child processes for heavy jobs	Kill when finished to release memory
Vector Search	Use memory-mapped files for embeddings	Only index hot queries in RAM
Audit & Alerts	Circular buffer in memory	Older entries stored in DB
SSOT Keys / Secrets	Load on demand	Never load all secrets at once


⸻

3. Node.js / Next.js Optimization Tips
	1.	Use next/server API routes efficiently
	•	Avoid large synchronous operations in route handlers.
	•	Offload heavy work to background workers.
	2.	Use Bun or Node 20+
	•	Bun is extremely memory-efficient for small services (~5-10MB per route handler).
	3.	Set memory limits per worker

node --max-old-space-size=200 worker.js


	4.	Stream logs and AI outputs

const stream = new ReadableStream({
  async start(controller) {
    for await (const chunk of source) {
      controller.enqueue(chunk);
    }
    controller.close();
  }
});


	5.	Avoid loading full datasets in memory
	•	Load files/embeddings on demand.
	•	Use sqlite, lmdb, or rocksdb as low-RAM storage.
